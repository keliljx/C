#include <stdio.h>

int main()
{
    char a[18];
 int i,k,last,sum=0,flag=1;
 gets(a);
 for(i=0;i<17;i++)
 {
  sum+=(a[i]-'0')*(i+1);
 }
 
 k=sum%11;

 if(a[17]=='Y')
  last=10;
 else
  last=a[17]-'0';

 for(i=0;i<17;i++) 
 {
  if(a[i]<'0'||a[i]>'9')
  {
   flag=0;
   break;
  }
 }
 if(k!=last) 
  flag=0;
 if(flag==1)
  printf("Yes\n");
 else 
  printf("No\n");

 }
