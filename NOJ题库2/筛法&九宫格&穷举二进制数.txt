void calculate();
void output();


int main(void){
	
	init();
	calculate();
	output();
	return 0;
}

void init(){
	for(int i=2;i<101;i++)
	{
		a[i] = 1;
	}
}

void calculate(){
	for(int i=2;i<101;i++){
		if(a[i]==1){
			for(int j=i+i;j<101;j+=i){
				a[j]=0;
			}
		}
	}
}

void output(){
	for(int i=2;i<101;i++){
		if(a[i]==1){
			printf("%d\n",i);
		}
	}
}

{
	int j,length;char temp;

	{
		temp=string[j];
		string[j]=string[length-1-j];
